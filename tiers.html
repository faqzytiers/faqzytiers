'' : 's'} tracked` : 'No tier results available yet.'; const modeChips = tiers.length ? tiers.map(renderModeChip).join('') : 'No tiers yet'; return `

${ign}

${region}
${best.tier}

${description}

${modeChips}
Top mode: ${best.gamemode} Best tier: ${best.tier}
`; } async function loadTierData() { try { const { data, error } = await supabaseClient .from('players') .select('ign, name, username, region, tiers'); if (error) { throw error; } if (!data || !data.length) { tierSubtitle.textContent = 'No tier data found.
